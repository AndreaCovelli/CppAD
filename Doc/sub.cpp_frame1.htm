Binary Subtraction: Example and Test


# include <CppAD/CppAD.h>
# include "NearEqualExt.h"
# include <cmath>

bool Sub(void)
{    bool ok = true;

     using namespace CppAD;

     // independent variable vector
     double u0 = .5;
     CppADvector< AD<double> > U(1);
     U[0]      = u0;
     Independent(U);

     AD<double> a = 2. * U[0] - 1.; // AD<double> - double
     AD<double> b = a  - 2;         // AD<double> - int
     AD<double> c = 3. - b;         // double     - AD<double> 
     AD<double> d = 4  - c;         // int        - AD<double> 

     // dependent variable vector 
     CppADvector< AD<double> > Z(1);
     Z[0] = U[0] - d;          // AD<double> - AD<double>

     // create f: U -> Z and vectors used for derivative calculations
     ADFun<double> f(U, Z); 
     CppADvector<double> v(1);
     CppADvector<double> w(1);

     // check value 
     ok &= NearEqual(Value(Z[0]) , u0-4+3-2*u0+1+2,  1e-10 , 1e-10);

     // forward computation of partials w.r.t. u
     size_t j;
     size_t p     = 5;
     double jfac  = 1.;
     double value = -1.;
     v[0]         = 1.;
     for(j = 1; j < p; j++)
     {    jfac *= j;
          w     = f.Forward(j, v); 
          ok &= NearEqual(jfac*w[0], value, 1e-10 , 1e-10); // d^jz/du^j
          v[0]  = 0.;
          value = 0.;
     }

     // reverse computation of partials of Taylor coefficients
     CppADvector<double> r(p); 
     w[0]  = 1.;
     r     = f.Reverse(p, w);
     jfac  = 1.;
     value = -1.;
     for(j = 0; j < p; j++)
     {    ok &= NearEqual(jfac*r[j], value, 1e-10 , 1e-10); // d^jz/du^j
          jfac *= (j + 1);
          value = 0.;
     }

     return ok;
}
