bool Run(bool TestOk(void), const char *name)
{    bool ok = true;
     using namespace std;

     ok &= TestOk();

     if( ok )
          std::cout << "Ok:    " << name << std::endl;
     else std::cout << "Error: " << name << std::endl;

     return ok;
}
