AD Comparison Operators: Example and Test


# include <CppAD/CppAD.h>

bool Compare(void)
{    bool ok = true;

     using namespace CppAD;

     AD<double> x = 3.;
     AD<double> y = 4.;
     AD<double> z = 4.;

     // AD<double> <op> AD<double>
     ok &= (y >  x);   
     ok &= (y >= x);   
     ok &= (x <  y);   
     ok &= (x <= y);   
     ok &= (x != y);   
     ok &= (z == y);   

     // double <op> AD<double>
     ok &= (3.5 >  x);   
     ok &= (3.5 >= x);   
     ok &= (3.5 <  y);   
     ok &= (3.5 <= y);   
     ok &= (3.5 != y);   
     ok &= (4.  == y);   

     // int <op> AD<double>
     ok &= (4 >  x);   
     ok &= (3 >= x);   
     ok &= (3 <  y);   
     ok &= (3 <= y);   
     ok &= (3 != y);   
     ok &= (4 == y);   

     // AD<double> <op> double
     ok &= (x <  3.5);   
     ok &= (x <= 3.5);   
     ok &= (y >  3.5);   
     ok &= (y >= 3.5);   
     ok &= (y != 3.5);   
     ok &= (y == 4.);   

     // AD<double> <op> int
     ok &= (x <  4);   
     ok &= (x <= 3);   
     ok &= (y >  3);   
     ok &= (y >= 3);   
     ok &= (y != 3);   
     ok &= (y == 4);   

     return ok;
}
